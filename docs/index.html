Instalación Rápida

# Clonar repositorio
git clone https://github.com/lucianoba77/sistema-gestion-cursos.git
cd sistema-gestion-cursos

# Instalar dependencias
composer install
npm install

# Configurar entorno
cp .env.example .env
php artisan key:generate

# Configurar base de datos
php artisan migrate
php artisan db:seed

# Iniciar servidor
php artisan serve
